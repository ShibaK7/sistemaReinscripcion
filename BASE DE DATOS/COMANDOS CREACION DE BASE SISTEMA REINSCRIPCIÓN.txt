/* CREACION DE TABLAS SIN LLAVE FORANEA */

/* TABLA ADMINISTRADOR */

CREATE TABLE ADMINISTRADOR (IDADMINISTRADOR VARCHAR(50) NOT NULL, ADNOMBRE VARCHAR(50) NOT NULL,
ADPATERNO VARCHAR(50) NOT NULL, ADMATERNO VARCHAR(50) NOT NULL, ADCONTRASEÑA VARCHAR(50) NOT NULL, 
ADPRIMERINGRESO INT NOT NULL, PRIMARY KEY(IDADMINISTRADOR));

/* TABLA CARRERA */

CREATE TABLE CARRERA (NCARRERA INT NOT NULL, CLAVECARRERA VARCHAR(10) NOT NULL, PRIMARY KEY (NCARRERA));

/* TABLA ESTADO */

CREATE TABLE ESTADO (ECLAVE INT NOT NULL, EESTADO VARCHAR(40) NOT NULL, PRIMARY KEY(ECLAVE));

/* TABLA ESTATUS */

CREATE TABLE ESTATUS(NSTATUS INT NOT NULL,ESTATUS VARCHAR(50) NOT NULL);

/* TABLA LISTAERRORES */

CREATE TABLE LISTAERRORES(CLAVEPROBLEMA INT NOT NULL, DESCRIPCION VARCHAR(300) NOT NULL, SOLUCION VARCHAR(300) 
NOT NULL, PRIMARY KEY (CLAVEPROBLEMA));

/* TABLA NIVEL */

CREATE TABLE NIVEL(NCLAVE INT NOT NULL, NNIVEL VARCHAR(40) NOT NULL, PRIMARY KEY (NCLAVE));

/* TABLA PROFESOR */

CREATE TABLE PROFESOR(IDTUTOR VARCHAR(50) NOT NULL, TNOMBRE VARCHAR(50) NOT NULL, TPATERNO VARCHAR(50) NOT NULL, 
TMATERNO VARCHAR(50) NOT NULL, TCONTRASENA VARCHAR(50) NOT NULL, PPRIMERINGRESO INT NOT NULL, PRIMARY KEY (IDTUTOR));

/* CREACION DE TABLAS CON LLAVE FORANEA */

/* TABLA ALUMNO */

CREATE TABLE ALUMNO (AMATRICULA VARCHAR(40) NOT NULL, ANOMBRE VARCHAR(40) NOT NULL, AAPATERNO VARCHAR(40) NOT NULL, 
AAMATERNO VARCHAR(40) NOT NULL, ACONTRASENA VARCHAR(40) NOT NULL, ACREDITOS INT NOT NULL, APRIMERINGRESO INT NOT NULL, 
NCARRERA INT NOT NULL, ESTADO INT NOT NULL, TASIGNADO INT NOT NULL, APROYECCION INT NOT NULL, AMAPA INT NOT NULL, 
PRIMERMAPA INT NOT NULL, PRIMARY KEY (AMATRICULA), FOREIGN KEY (NCARRERA) REFERENCES CARRERA (NCARRERA));

/* TABLA MATERIA */

CREATE TABLE MATERIA (MCLAVE INT NOT NULL, MNOMBRE VARCHAR(50) NOT NULL, CREDITOS INT NOT NULL, NCLAVE INT NOT NULL, 
NCARRERA INT NOT NULL, PRIMARY KEY (MCLAVE), FOREIGN KEY (NCLAVE) REFERENCES NIVEL(NCLAVE));

/* TABLA CLAVESMATERIA */

CREATE TABLE CLAVESMATERIA(MCLAVE INT NOT NULL, CLAVEREAL VARCHAR(40) NOT NULL, FOREIGN KEY (MCLAVE) 
REFERENCES MATERIA(MCLAVE));

/* TABLA PROBLEMA */

CREATE TABLE PROBLEMA (CLAVEPROBLEMA INT NOT NULL, IDTUTOR VARCHAR(50) NOT NULL, AMATRICULA VARCHAR(50) NOT NULL,
FOREIGN KEY (CLAVEPROBLEMA) REFERENCES LISTAERRORES(CLAVEPROBLEMA), FOREIGN KEY (IDTUTOR) REFERENCES PROFESOR(IDTUTOR), 
FOREIGN KEY (AMATRICULA) REFERENCES ALUMNO (AMATRICULA));

/ *TABLA PROYECCIÓN */

CREATE TABLE PROYECCION(AMATRICULA VARCHAR(40) NOT NULL, MCLAVE INT NOT NULL, FOREIGN KEY (AMATRICULA) REFERENCES 
ALUMNO(AMATRICULA), FOREIGN KEY (MCLAVE) REFERENCES MATERIA(MCLAVE));

/* TABLA TUTOR */

CREATE TABLE TUTOR (IDTUTOR VARCHAR(50) NOT NULL, AMATRICULA VARCHAR(40) NOT NULL, BLOQUE VARCHAR(40) NOT NULL,
FOREIGN KEY (IDTUTOR) REFERENCES PROFESOR (IDTUTOR), FOREIGN KEY (AMATRICULA) REFERENCES ALUMNO (AMATRICULA));

/* TABLA HISTORIAL*/

CREATE TABLE HISTORIAL (MCLAVE INT NOT NULL, AMATRICULA VARCHAR(40) NOT NULL, ECLAVE INT NOT NULL, NUMOPTATIVA 
INT NOT NULL, FOREIGN KEY (MCLAVE) REFERENCES MATERIA(MCLAVE), FOREIGN KEY (AMATRICULA) REFERENCES ALUMNO (AMATRICULA), 
FOREIGN KEY (ECLAVE) REFERENCES ESTADO(ECLAVE));
